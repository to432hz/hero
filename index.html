.filter(it => domainsFilter.length ? domainsFilter.every(d=>it.domain.includes(d)) : true)

.filter(it => {

})

.sort((a,b)=>{

return +new Date(b.updatedAt) - +new Date(a.updatedAt);

});

},[items,query,status,domainsFilter,sortKey]);
